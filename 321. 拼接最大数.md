[拼接最大数](https://leetcode-cn.com/problems/create-maximum-number/)

```js
/**
 * @param {number[]} nums1
 * @param {number[]} nums2
 * @param {number} k
 * @return {number[]}
 */
var maxNumber = function (nums1, nums2, k) {
	const len1 = nums1.length;
	const len2 = nums2.length;
	let max = [];
	for (let i = 0; i <= k; i++) {
		if (i <= len1 && k - i <= len2) {
			const p1 = pick(nums1, i);
			const p2 = pick(nums2, k - i);
			const p3 = merge(p1, p2);
			max = pickMax(max, p3);
		}
	}
	return max;

	function pick(arr, n) {
		if (n === 0) {
			return [];
		}
		const len = arr.length;
		if (len === n) {
			return arr;
		}
		const stack = [];
		let k = len - n;
		for (let v of arr) {
			while (stack[stack.length - 1] < v && k > 0) {
				stack.pop();
				k--;
			}
			stack.push(v);
		}
		return stack.slice(0, n);
	}
	function pickMax(arr1, arr2) {
		const len1 = arr1.length;
		const len2 = arr2.length;

		if (len1 === len2) {
			if (arr1.join('') > arr2.join('')) {
				return arr1;
			} else {
				return arr2;
			}
		} else {
			if (len1 > len2) {
				return arr1;
			} else {
				return arr2;
			}
		}
	}
	function merge(arr1, arr2) {
		let i = 0;
		let j = 0;
		const stack = [];
		while (i <= arr1.length || j <= arr2.length) {
			const a = arr1[i];
			const b = arr2[j];

			if (i === arr1.length) {
				stack.push(...arr2.slice(j));
				break;
			}
			if (j === arr2.length) {
				stack.push(...arr1.slice(i));
				break;
			}
			if (a === b) {
				if (arr1.slice(i).join('') > arr2.slice(j).join('')) {
					stack.push(a);
					i++;
				} else {
					stack.push(b);
					j++;
				}
			} else if (a > b) {
				stack.push(a);
				i++;
			} else {
				stack.push(b);
				j++;
			}
		}
		return stack;
	}
};
```
