### 移掉K位数字
```js
/**
 * @param {string} num
 * @param {number} k
 * @return {string}
 */
var removeKdigits = function (num, k) {
	const stack = [];
	const remain = num.length - k;
	for (let v of num) {
		while (stack.length && stack[stack.length - 1] > v && k > 0) {
			stack.pop();
			k--;
		}
		stack.push(v);
	}
	return stack.slice(0, remain).join('').replace(/^0+/, '') || '0';
};
```
